#include <algorithm>
using namespace std;
#define MAX_N	100


int n, m;
bool states[MAX_N];
int pri[MAX_N];
int ranks[MAX_N];


int calc()
{
	sort(ranks, ranks + n);
	int curr = 0, top = n - 1, steps = 0;

	while (top >= 0) {
		if (!states[curr] && ranks[top] == pri[curr]) {
			states[curr] = true;
			--top;
			++steps;
			if (curr == m) break;
		}
		curr = (curr + 1) % n;
	}

	return steps;
}


int main()
{
	int t; scanf_s("%d", &t);
	for (int i = 0; i < t; ++i) {
		scanf_s("%d %d", &n, &m);
		for (int j = 0; j < n; ++j) {
			scanf_s("%d", &pri[j]);
			ranks[j] = pri[j];
			states[j] = false;
		}
		printf("%d\n", calc());
	}
    return 0;
}
