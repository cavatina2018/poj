#include <math.h>
#define SCANF	scanf


int main()
{
	while (true)
	{
		int b, n;
		SCANF("%d %d", &b, &n);
		if (!b)
		{
			break;
		}

		int a = (int)pow((double)b, 1.0 / n);
		int p1 = (int)pow((double)a, n);
		int p2 = (int)pow((double)(a + 1), n);
		if (b - p1 < p2 - b)
		{
			printf("%d\n", a);
		}
		else
		{
			printf("%d\n", a + 1);
		}
	}
    return 0;
}
