#define MAX_N	4500
#define M		1000000007


int n, m;
int results[MAX_N + 1][MAX_N + 1];


int calc()
{
	for (int j = 0; j <= m; ++j) {
		results[1][j] = 1;
	}

	for (int i = 2; i <= n; ++i) {
		results[i][0] = 1;
		for (int j = 1; j <= m; ++j) {
			results[i][j] = results[i - 1][j];
			//results[i][j] %= M;
			if (j >= i) {
				results[i][j] += results[i][j - i];
				results[i][j] %= M;
			}
		}
	}

	return results[n][m];
}


int main()
{
	scanf_s("%d %d", &n, &m);
	printf("%d\n", calc());
    return 0;
