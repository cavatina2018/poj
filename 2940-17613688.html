#include <math.h>
#define MAX_N	100000


int n;
int a[MAX_N];


long long calc()
{
	long long res = 0;

	for (int i = 0; i < n - 1; ++i) {
		res += abs(a[i]);
		a[i + 1] += a[i];
	}
	return res;
}


int main()
{
	while (true) {
		scanf_s("%d", &n);
		if (!n) break;

		for (int i = 0; i < n; ++i) {
			scanf_s("%d", &a[i]);
		}

		printf("%lld\n", calc());
	}
    return 0;
}
