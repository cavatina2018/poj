import java.util.*;
import java.io.*;
import java.lang.*;


public class Main {
	
	public static boolean[][] states = new boolean[1001][101];
	public static BigInteger[][] results = new BigInteger[1001][101];
	
	public static BigInteger calc(int n, int k)
	{
		if (n == 0 || k == 1) {
			return BigInteger.ONE;
		}

		if (states[n][k]) {
			return results[n][k];
		}

		BigInteger res = calc(n, k - 1);
		if (n >= k) {
			res = res.add(calc(n - k, k));
		}

		results[n][k] = res;
		states[n][k] = true;
		return res;
	}
	
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);
		
		int n = in.nextInt(), k = in.nextInt();
		if(k > n) k = n;
		System.out.println(calc(n, k));
	}
};
