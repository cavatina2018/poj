double Round(double val)
{
	val = (val> 0.0) ? floor(val + 0.5) : ceil(val - 0.5);
	return val;
}


int FC(double a, double b)
{
	if (a > b + 0.000001) return 1;
	if (a < b - 0.000001) return -1;

	long long a1 = (long long)(Round(a * 10000000));
	long long b1 = (long long)(Round(b * 10000000));

	if (a1 > b1) return 1;
	if (a1 < b1) return -1;
	else return 0;
}


bool test(long long count)
{
	double angle = PI / count;
	double dist = (a - b) * sin(angle) * 2;
	return FC(dist, c + 2 * b) >= 0;
}


long long calc()
{
	if (FC(a, b) < 0) return 0;
	a /= 2;
	b /= 2;

	long long res = 1, x = 2, y = INT_MAX;
	while (x <= y) {
		long long z = (x + y) / 2;
		if (test(z)) {
			if (z > res) res = z;
			x = z + 1;
		}
		else {
			y = z - 1;
		}
	}
	return res;
}
