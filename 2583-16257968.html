using namespace std;


void calc(int x, int y, int z)
{
	int c = x;
	int a = ((z - x) - 2 * (y - x)) / 2;
	int b = y - x - a;

	x = a * 9 + b * 3 + c;
	y = a * 16 + b * 4 + c;
	z = a * 25 + b * 5 + c;
	cout << x << " " << y << " " << z << endl;
}


int main()
{
	while (true)
	{
		int x, y, z;
		if (!(cin >> x >> y >> z))
		{
			break;
		}
		calc(x, y, z);
	}
    return 0;
}
