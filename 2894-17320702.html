#include <stdlib.h>


int n;
int starts[26];
int ends[26];
int m, M;


int main()
{
	int t; scanf_s("%d", &t);
	for (int i = 0; i < t; ++i) {
		scanf_s("%d", &n);

		m = INT_MAX, M = 0;
		for (int j = 0; j < n; ++j) {
			char ch[2];
			scanf_s("%s %d %d", ch, 2, &starts[j], &ends[j]);
			if (starts[j] < m) m = starts[j];
			if (ends[j] > M) M = ends[j];
		}

		for (int j = m; j < M; ++j) {
			int count = 0;
			for (int k = 0; k < n; ++k) {
				if (j >= starts[k] && j < ends[k]) {
					++count;
				}
			}
			if(count > 0) putchar((char)('A' + count - 1));
		}
		printf("\n");
	}
    return 0;
}
