#define MAX_D	500
#define MAX_N	100


int n, d;
int states[MAX_D][MAX_N];


bool calc()
{
	for (int i = 0; i < n; ++i) {
		bool flag = true;
		for (int j = 0; j < d; ++j) {
			if (!states[j][i]) {
				flag = false;
				break;
			}
		}
		if (flag) return true;
	}
	return false;
}


int main()
{
	while (true) {
		scanf_s("%d %d", &n, &d);
		if (!n && !d) break;

		for (int i = 0; i < d; ++i) {
			for (int j = 0; j < n; ++j) {
				scanf_s("%d", &states[i][j]);
			}
		}
		if (calc()) {
			printf("yes\n");
		}
		else {
			printf("no\n");
		}
	}
    return 0;
}
