#include <queue>
#include <algorithm>
using namespace std;
#define MAX_N	5000000
#define MAX_M	10


int n1, n2, n;
int great[MAX_M];
int least[MAX_M];


void add_great(int v)
{
	for (int i = 0; i < n1; ++i) {
		if (great[i] < v) {
			for (int j = n1 - 1; j > i; --j) {
				great[j] = great[j - 1];
			}
			great[i] = v;
			return;
		}
	}
}


void add_least(int v)
{
	for (int i = 0; i < n2; ++i) {
		if (least[i] > v) {
			for (int j = n2 - 1; j > i; --j) {
				least[j] = least[j - 1];
			}
			least[i] = v;
			return;
		}
	}
}


int main()
{
	while (true) {
		scanf_s("%d %d %d", &n1, &n2, &n);
		if (!n1 && !n2 && !n) break;

		double sum = 0.0;
		fill(great, great + n1, 0);
		fill(least, least + n2, INT_MAX);

		for (int i = 0; i < n; ++i) {
			int v; scanf_s("%d", &v);
			sum += v;
			add_great(v);
			add_least(v);
		}

		for (int i = 0; i < n1; ++i) {
			sum -= great[i];
		}
		for (int i = 0; i < n2; ++i) {
			sum -= least[i];
		}
		printf("%.6lf\n", sum / (n - n1 - n2));
	}
	return 0;
}
