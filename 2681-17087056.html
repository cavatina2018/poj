#include <algorithm>
using namespace std;


char str[2][1024];
int counts[2][26];


int calc()
{
	memset(counts[0], 0, sizeof(counts[0]));
	memset(counts[1], 0, sizeof(counts[1]));

	for (int i = 0; i < 2; ++i) {
		for (int j = 0; str[i][j]; ++j) {
			counts[i][str[i][j] - 'a']++;
		}
	}

	int res = 0;
	for (int i = 0; i < 26; ++i) {
		if (counts[0][i] > counts[1][i]) {
			res += counts[0][i] - counts[1][i];
		}
		else {
			res += counts[1][i] - counts[0][i];
		}
	}
	return res;
}


int main()
{
	int t; scanf_s("%d", &t);
	gets_s(str[0], 1024);

	for (int i = 1; i <= t; ++i) {
		gets_s(str[0], 1024);
		gets_s(str[1], 1024);
		printf("Case #%d:  %d\n", i, calc());
	}
    return 0;
}
