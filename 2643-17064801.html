#include <string>
#include <map>
using namespace std;


map<string, string> parties;
map<string, int> votes;


int main()
{
	char buf[100];

	int n; cin >> n; cin.getline(buf, 100);
	for (int i = 0; i < n; ++i)
	{
		cin.getline(buf, 100);
		string name(buf);

		cin.getline(buf, 100);
		string party(buf);

		parties[name] = party;
	}

	int m; cin >> m; cin.getline(buf, 100);
	for (int i = 0; i < m; ++i)
	{
		cin.getline(buf, 100);
		string name(buf);

		votes[name]++;
	}

	int s = -1, count = 0;
	string c;

	for (map<string, int>::iterator it = votes.begin(); it != votes.end(); ++it)
	{
		if (it->second > s)
		{
			s = it->second;
			count = 1;
			c = it->first;
		}
		else if (it->second == s)
		{
			count++;
		}
	}

	if (count > 1)
	{
		cout << "tie" << endl;
	}
	else
	{
		cout << parties[c] << endl;
	}
    return 0;
}
