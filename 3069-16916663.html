#include <algorithm>
using namespace std;


#define MAX_N	1000
int N, R;
int X[MAX_N];


int calc(int index)
{
	if (index >= N)
	{
		return 0;
	}

	int i = index + 1;
	for (; i < N; ++i)
	{
		if (X[i] - X[index] > R)
		{
			break;
		}
	}

	--i;

	int j = i + 1;
	if (i > index)
	{		
		for (; j < N; ++j)
		{
			if (X[j] - X[i] > R)
			{
				break;
			}
		}
	}

	return 1 + calc(j);
}


int main()
{
	while (true)
	{
		cin >> R >> N;
		if (R < 0 || N < 0)
		{
			break;
		}

		for (int i = 0; i < N; ++i)
		{
			cin >> X[i];
		}
		sort(X, X + N);

		cout << calc(0) << endl;
	}
    return 0;
}
