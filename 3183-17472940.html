int main()
{
	int n; scanf_s("%d", &n);
	int prev = -1;
	bool flag = false;
	for (int i = 0; i < n; ++i) {
		int h; scanf_s("%d", &h);
		if (flag) {
			if (h <= prev) {
				printf("%d\n", i);
			}
			if (h < prev) {
				flag = false;
			}
		}
		else {
			if (h >= prev) {
				flag = true;
			}
		}
		prev = h;
	}
	if (flag) {
		printf("%d\n", n);
	}
    return 0;
}
