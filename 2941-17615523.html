#define MAX_N	1000


int n;
int matrix[MAX_N][MAX_N];


bool calc()
{
	for (int i = 1; i < n; ++i) {
		int d = matrix[i][0] - matrix[i - 1][0];
		for (int j = 1; j < n; ++j) {
			if (d != matrix[i][j] - matrix[i - 1][j]) {
				return false;
			}
		}
	}
	return true;
}


int main()
{
	while (true) {
		scanf_s("%d", &n);
		if (!n) break;

		for (int i = 0; i < n; ++i) {
			for (int j = 0; j < n; ++j) {
				scanf_s("%d", &matrix[i][j]);
			}
		}

		if (calc()) {
			printf("homogeneous\n");
		}
		else {
			printf("not homogeneous\n");
		}
	}
    return 0;
}
