string vowel(string& s)
{
	string res;

	for (size_t i = 0; i < s.length(); ++i) {
		if (s[i] >= 'A' && s[i] <= 'Z') {
			s[i] += 'a' - 'A';
		}
		if (s[i] == 'a' || s[i] == 'e' || s[i] == 'i' || s[i] == 'o' || s[i] == 'u') {
			res = s[i];
		}
	}

	if (res.length() == 0) {
		res = s;
	}

	return res;
}
