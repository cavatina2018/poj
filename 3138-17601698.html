#include <cstring>
#define MAX_N	100


bool states[MAX_N + 1];
bool conds[MAX_N + 1][3];


int main()
{
	for (int tcase = 1; ; ++tcase) {
		int n, t, m; scanf_s("%d %d %d", &n, &t, &m);
		if (!n && !t && !m) break;

		memset(states, 0, sizeof(states));
		memset(conds, 0, sizeof(conds));

		for (int i = 0; i < n; ++i) {
			int id, p, q; scanf_s("%d %d %d", &id, &p, &q);
			states[id] = true;
			if (p) conds[id][0] = true;
			if (q) conds[id][1] = true;;
		}

		for (int i = 0; i < t; ++i) {
			int id, count; scanf_s("%d %d", &id, &count);
			if (count >= m) conds[id][2] = true;
		}

		int sum = 0;
		for (int i = 1; i <= MAX_N; ++i) {
			if (!states[i]) continue;
			for (int j = 0; j < 3; ++j) {
				if (conds[i][j]) {
					sum++;
				}
			}
		}

		printf("Case %d: %d\n", tcase, sum);
	}

    return 0;
}
