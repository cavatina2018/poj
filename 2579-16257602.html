#include <string>
using namespace std;
#define MAX_R	10
#define MAX_C	10


int r, c;
int pic[MAX_R][MAX_C];
int res[MAX_R][MAX_C];


void blur()
{
	for (int i = 0; i < r - 1; ++i)
	{
		for (int j = 0; j < c - 1; ++j)
		{
			res[i][j] = (pic[i][j] + pic[i + 1][j] + pic[i][j + 1] + pic[i + 1][j + 1]) / 4;
		}
	}
}


int main()
{
	while (true)
	{
		string start, end;
		cin >> start;
		if (start == "ENDOFINPUT")
		{
			break;
		}

		cin >> r >> c;
		for (int i = 0; i < r; ++i)
		{
			string line; cin >> line;
			for (int j = 0; j < c; ++j)
			{
				pic[i][j] = line[j] - '0';
			}
		}

		blur();

		for (int i = 0; i < r - 1; ++i)
		{
			string line;
			for (int j = 0; j < c - 1; ++j)
			{
				line += ('0' + res[i][j]);
			}
			cout << line << endl;
		}

		cin >> end;
	}
    return 0;
}
