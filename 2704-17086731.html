#include <string>
using namespace std;
#define MAX_N	34

int n;
int matrix[MAX_N][MAX_N];
bool states[MAX_N][MAX_N];
long long paths[MAX_N][MAX_N];


long long calc(int i, int j)
{
	if (states[i][j]) return paths[i][j];

	long long res = 0;
	if (i == n - 1 && j == n - 1) {
		res = 1;
	}
	else {
		int step = matrix[i][j];
		if (step && i + step < n) {
			res += calc(i + step, j);
		}
		if (step && j + step < n) {
			res += calc(i, j + step);
		}
	}

	paths[i][j] = res;
	states[i][j] = true;
	return res;
}


int main()
{
	while (true) {
		cin >> n;
		if (n <= 0) break;

		for (int i = 0; i < n; ++i) {
			string s; cin >> s;
			for (int j = 0; j < n; ++j) {
				matrix[i][j] = s[j] - '0';
			}
		}

		memset(states, 0, sizeof(states));
		cout << calc(0, 0) << endl;
	}
    return 0;
}
