#include <iostream>
using namespace std;


int rounder(int n)
{
	int a = 10, b = 5;

	while (true)
	{
		if (n <= a)
		{
			break;
		}

		if (n % a >= b)
		{
			n = (n / a + 1) * a;
		}
		else
		{
			n = n / a * a;
		}

		a *= 10, b *= 10;
	}

	return n;
}


int main()
{
	int t;
	cin >> t;
	for (int i = 0; i < t; ++i)
	{
		int n;
		cin >> n;
		cout << rounder(n) << endl;
	}
    return 0;
}
