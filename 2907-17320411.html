#include <algorithm>
using namespace std;


int b;
int x[10], y[10];
int startx, starty;
int n, m;


int dist(int x1, int y1, int x2, int y2)
{
	return abs(x1 - x2) + abs(y1 - y2);
}


int calc()
{
	int perm[10];
	for (int i = 0; i < b; ++i) {
		perm[i] = i;
	}

	int res = INT_MAX;

	do {
		int ret = dist(startx, starty, x[perm[0]], y[perm[0]]);
		for (int i = 0; i < b - 1; ++i) {
			ret += dist(x[perm[i]], y[perm[i]], x[perm[i + 1]], y[perm[i + 1]]);
		}
		ret += dist(x[perm[b - 1]], y[perm[b - 1]], startx, starty);
		if (ret < res) res = ret;
	} while (next_permutation(perm, perm + b));

	return res;
}


int main()
{
	int t; scanf_s("%d", &t);
	for (int i = 0; i < t; ++i) {
		scanf_s("%d %d %d %d %d", &n, &m, &startx, &starty, &b);
		for (int j = 0; j < b; ++j) {
			scanf_s("%d %d", &x[j], &y[j]);
		}
		printf("The shortest path has length %d\n", calc());
	}
    return 0;
}
