#include <queue>
using namespace std;

struct Less
{
	bool operator()(int a, int b)
	{
		return a > b;
	}
};

int main()
{
	priority_queue<int, vector<int>, Less> q;
	int n; scanf_s("%d", &n);
	for (int i = 0; i < n; ++i) {
		int l; scanf_s("%d", &l);
		q.push(l);
	}

	long long cost = 0;
	while (q.size() > 1) {
		int a = q.top(); q.pop();
		int b = q.top(); q.pop();
		cost += a + b;
		q.push(a + b);
	}

	printf("%lld\n", cost);
    return 0;
}
