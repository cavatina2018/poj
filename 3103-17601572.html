int main()
{
	double x, y, z;
	int n;
	scanf_s("%lf %lf %lf %d", &x, &y, &z, &n);

	double d = x / n, s = 0.0;
	for (int i = 0; i < n; ++i) {
		printf("%.8f %.8f %.8f ", s, 0.0, 0.0);
		printf("%.8f %.8f %.8f\n", s + d, y, z);
		s += d;
	}
    return 0;
}
