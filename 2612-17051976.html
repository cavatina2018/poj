#include <string>
using namespace std;
#define MAX_N	10

int n;
char mine[MAX_N][MAX_N];
char touch[MAX_N][MAX_N];


void addmine(int& count, int i, int j)
{
	if (i < 0 || i >= n || j < 0 || j >= n)
	{
		return;
	}
	if (mine[i][j] == '*')	++count;
}


int main()
{
	bool flag = false;

	cin >> n;
	for (int i = 0; i < n; ++i)
	{
		string s; cin >> s;
		for (int j = 0; j < n; ++j)
		{
			mine[i][j] = s[j];
		}
	}
	for (int i = 0; i < n; ++i)
	{
		string s; cin >> s;
		for (int j = 0; j < n; ++j)
		{
			touch[i][j] = s[j];
			if (touch[i][j] == 'x' && mine[i][j] == '*')
			{
				flag = true;
			}
		}
	}

	for (int i = 0; i < n; ++i)
	{
		string s;
		for (int j = 0; j < n; ++j)
		{
			if (flag && mine[i][j] == '*')
			{
				s += '*';
			}
			else if (touch[i][j] == '.')
			{
				s += '.';
			}
			else
			{
				int count = 0;
				addmine(count, i, j - 1);
				addmine(count, i, j + 1);
				addmine(count, i - 1, j);
				addmine(count, i + 1, j);
				addmine(count, i - 1, j - 1);
				addmine(count, i + 1, j + 1);
				addmine(count, i - 1, j + 1);
				addmine(count, i + 1, j - 1);

				s += (char)('0' + count);
			}
		}
		cout << s << endl;
	}
    return 0;
}
