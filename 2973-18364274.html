#include <cstring>
#include <algorithm>
using namespace std;
#define MAX_N	1000

int n, m;
char dict[MAX_N][8];
char pattern[8];
int perm[8];


bool test(int idx)
{
	if (strlen(dict[idx]) > m) return false;

	for (int i = 0; i < m; ++i) {
		perm[i] = i;
	}

	do {
		bool flag = true;
		for (int i = 0; dict[idx][i]; ++i) {
			if (pattern[perm[i]] != '_' && pattern[perm[i]] != dict[idx][i]) {
				flag = false;
				break;
			}
		}
		if (flag) return true;
	} while (next_permutation(perm, perm + m));

	return false;
}


int calc()
{
	int res = 0;

	m = strlen(pattern);
	for (int i = 0; i < n; ++i) {
		if (test(i)) ++res;
	}

	return res;
}


int main()
{
	while (true) {
		scanf_s("%d", &n);
		if (!n) break;

		for (int i = 0; i < n; ++i) {
			scanf_s("%s", dict[i], 8);
		}
		scanf_s("%s", pattern, 8);

		printf("%d\n", calc());
	}
    return 0;
