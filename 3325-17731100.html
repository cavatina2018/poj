#include <stdlib.h>


int main()
{
	while (true) {
		int n; scanf_s("%d", &n);
		if (!n) break;

		int s = 0, m = INT_MAX, M = INT_MIN;
		for (int i = 0; i < n; ++i) {
			int score; scanf_s("%d", &score);
			s += score;
			if (score > M) M = score;
			if (score < m) m = score;
		}

		printf("%d\n", (s - M - m) / (n - 2));
	}
    return 0;
}
