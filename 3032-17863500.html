#include <cstring>
#define MAX_N	13


int n;
int cards[MAX_N];
bool states[MAX_N];


void calc()
{
	memset(states, 0, sizeof(states));

	for (int i = 1, start = 0; i <= n; ++i) {
		for (int j = 0; j < i; ++j) {
			while (states[start]) {
				start = (start + 1) % n;
			}
			start = (start + 1) % n;
			while (states[start]) {
				start = (start + 1) % n;
			}
		}
		cards[start] = i;
		states[start] = true;
	}

	for (int i = 0; i < n; ++i) {
		printf("%d ", cards[i]);
	}
	printf("\n");
}


int main()
{
	int t; scanf_s("%d", &t);
	for (int i = 0; i < t; ++i) {
		scanf_s("%d", &n);
		calc();
	}
    return 0;
}
