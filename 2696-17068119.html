using namespace std;


#define MAX_N	1000
bool states[MAX_N + 1];
int results[MAX_N + 1];
int a, b, c, d, e, f, g, h;


int mod(int n, int d)
{
	if (n % d == 0) return 0;
	if (n >= 0) return n % d;
	return d - (-n) % d;
}


int calc(int n)
{
	if (n == 0) return a;
	if (n == 1) return b;
	if (n == 2) return c;

	if (states[n]) return results[n];

	int res = 0;
	if (n % 2)
	{
		res = mod(d * calc(n - 1) + e * calc(n - 2) - f * calc(n - 3), g);
	}
	else
	{
		res = mod(f * calc(n - 1) - d * calc(n - 2) + e * calc(n - 3), h);
	}

	states[n] = true;
	results[n] = res;
	return res;
}


int main()
{
	int t; cin >> t;
	for (int i = 0; i < t; ++i)
	{
		int n; cin >> a >> b >> c >> d >> e >> f >> g >> h >> n;
		memset(states, 0, sizeof(states));
		cout << calc(n) << endl;
	}
    return 0;
}
