#include <vector>
using namespace std;
#define MAX_K	200
#define MAX_N	2000


int k, n, m;
int cows[MAX_N];
vector<int> paths[MAX_N];
bool states[MAX_N];


int dfs(int node)
{
	int res = cows[node];
	states[node] = true;

	for (size_t i = 0; i < paths[node].size(); ++i) {
		int next = paths[node][i];
		if (states[next]) continue;
		res += dfs(next);
	}
	return res;
}


int calc()
{
	int res = 0;
	for (int i = 1; i <= n; ++i) {
		memset(states, 0, sizeof(states));
		if (dfs(i) >= k) {
			++res;
		}
	}
	return res;
}


int main()
{
	scanf_s("%d %d %d", &k, &n, &m);
	for (int i = 0; i < k; ++i) {
		int e; scanf_s("%d", &e);
		cows[e]++;
	}
	for (int i = 0; i < m; ++i) {
		int a, b; scanf_s("%d %d", &a, &b);
		paths[b].push_back(a);
	}
	printf("%d\n", calc());
    return 0;
}
