#include <math.h>
#include <stdlib.h>
#include <cstring>


int n1, n2, n3, r;


int calc()
{
	int t = (int)ceil(100.0 / (r + 2));

	int res = INT_MAX;
	for (int i = t; i >= 0; --i) {
		int ret1 = i, r1 = 100 - i * (r + 2);
		if (r1 > 0) {
			ret1 += (int)ceil((double)r1 / r);
		}
		ret1 *= n1;

		int ret2 = i, r2 = 100 - i * (r - 1);
		if (r2 > 0) {
			ret2 += (int)ceil((double)r2 / r);
		}
		ret2 *= n2;

		int ret3 = n3 * (int)ceil(100.0 / r);
		if (ret1 + ret2 + ret3 < res) {
			res = ret1 + ret2 + ret3;
		}
	}
	return res;
}


int main()
{
	while (true) {
		scanf_s("%d %d %d %d", &n1, &n2, &n3, &r);
		if (!n1 && !n2 && !n3 && !r) break;

		printf("%d\n", calc());
	}
    return 0;
}
