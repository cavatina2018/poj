#include <string.h>
#define MAX_N	80


int n, count;
char buffer[MAX_N];


int main()
{
	while (true) {
		scanf_s("%s", buffer, MAX_N);
		if (!strcmp(buffer, "#")) break;

		n = strlen(buffer), count = 0;

		for (int i = 0; i < n; ++i) {
			if (buffer[i] == '.') {
				count += 2;
			}
			else if (buffer[i] == '/') {
				bool flag = true;
				for (int j = i - 1; j >= 0; --j) {
					if (buffer[j] == '.') {
						break;
					}
					if (buffer[j] == '/' || buffer[j] == '\\' || buffer[j] == '|') {
						flag = false;
						break;
					}
				}
				if (flag) count += 2;
			}
			else if (buffer[i] == '\\') {
				bool flag = true;
				for (int j = i + 1; j < n; ++j) {
					if (buffer[j] == '.') {
						break;
					}
					if (buffer[j] == '/' || buffer[j] == '\\' || buffer[j] == '|') {
						flag = false;
						break;
					}
				}
				if (flag) count += 2;
			}
			else if (buffer[i] == '|') {
				{
					bool flag = true;
					for (int j = i - 1; j >= 0; --j) {
						if (buffer[j] == '.') {
							break;
						}
						if (buffer[j] == '/' || buffer[j] == '\\' || buffer[j] == '|') {
							flag = false;
							break;
						}
					}
					if (flag) count += 1;
				}
				{
					bool flag = true;
					for (int j = i + 1; j < n; ++j) {
						if (buffer[j] == '.') {
							break;
						}
						if (buffer[j] == '/' || buffer[j] == '\\' || buffer[j] == '|') {
							flag = false;
							break;
						}
					}
					if (flag) count += 1;
				}
			}
		}

		printf("%d\n", (int)(100.0 * count / 2 / n));
	}
    return 0;
}
