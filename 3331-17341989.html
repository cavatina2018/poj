    • import java.math.*;
import java.util.*;
import java.io.*;
import java.lang.*;


public class Main {
	static BigInteger[] fact = new BigInteger[367];
	static int[][] counts = new int[367][10];
		
	static void init()
	{
		fact[1] = BigInteger.ONE;
		counts[1][1] = 1;
		
		for(int i = 2; i <= 366; ++i) {
			fact[i] = fact[i - 1].multiply(BigInteger.valueOf(i));
			
			BigInteger t = fact[i];
			while(t.compareTo(BigInteger.ZERO) > 0) {
				int digit = t.mod(BigInteger.valueOf(10)).intValue();
				counts[i][digit]++;
				t = t.divide(BigInteger.valueOf(10));
			}
		}
	}
	
    public static void main(String[] args) {
		init();
        Scanner in = new Scanner(System.in);
		int t = in.nextInt();
		for(int i = 0; i < t; ++i) {
			int y = in.nextInt();
			int d = in.nextInt();
			System.out.println(counts[y][d]);
		}
	}
};
