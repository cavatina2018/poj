#include <iostream>
using namespace std;


int max_factor(int num)
{
	int res = 0;

	for (int i = 2; i <= sqrt((double)num); )
	{
		if (num % i == 0)
		{
			num /= i;
			if (i > res)
			{
				res = i;
			}
		}
		else
		{
			++i;
		}
	}

	if (res == 0 || res < num)
	{
		res = num;
	}
	return res;
}


int main()
{
	int max_f = 0, max_num = 0;
	int n; cin >> n;

	for (int i = 0; i < n; ++i)
	{
		int num; cin >> num;
		int factor = max_factor(num);
		//cout << factor << endl;
		if (factor > max_f)
		{
			max_f = factor;
			max_num = num;
		}
	}
	cout << max_num << endl;
    return 0;
}
