char buffer[600000];


void calc(int level, int start, int len)
{
	if (level == 0) {
		buffer[start] = '-';
		return;
	}
	calc(level - 1, start, len / 3);
	calc(level - 1, start + len * 2 / 3, len / 3);
}

{

		int n;
