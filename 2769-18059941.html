#include <math.h>
#include <cstring>
#define MAX_N	300
#define MAX_S	1000000

int n;
int ids[MAX_N];
bool states[MAX_S + 1];


void add(int x)
{
	states[x] = true;

	for (int i = 2; i * i <= x; ++i) {
		if (x % i) continue;
		states[i] = states[x / i] = true;
	}
}


int calc()
{
	if (n <= 1) {
		return 1;
	}

	memset(states, 0, sizeof(states));

	for (int i = 0; i < n; ++i) {
		for (int j = i + 1; j < n; ++j) {
			add(abs(ids[i] - ids[j]));
		}
	}

	for (int i = 2; i <= MAX_S; ++i) {
		if (!states[i]) return i;
	}

	while (true) printf("...");
	return 0;
}



int main()
{
	int t; scanf_s("%d", &t);
	for(int j = 0; j < t; ++j) {
		scanf_s("%d", &n);
		while (n < 1 || n > MAX_N) printf("...");

		for (int i = 0; i < n; ++i) {
			scanf_s("%d", &ids[i]);
		}
		printf("%d\n", calc());
	}
    return 0;
}
