double X[4];
double Y[4];


bool Equal(double a, double b)
{
	return a > b - 0.000001 && a < b + 0.000001;
}


int find()
{
	for (int i = 0; i < 3; ++i) {
		for (int j = i + 1; j < 4; ++j) {
			if (Equal(X[i], X[j]) && Equal(Y[i], Y[j])) {
				return i;
			}
		}
	}
	return 0;
}


int main()
{
	while (scanf_s("%lf %lf %lf %lf %lf %lf %lf %lf", &X[0], &Y[0], &X[1], &Y[1], &X[2], &Y[2], &X[3], &Y[3]) >= 8) {
		int idx = find();
		double x = 0.0, y = 0.0;
		for (int i = 0; i < 4; ++i) {
			if (Equal(X[i], X[idx]) && Equal(Y[i], Y[idx])) {
				continue;
			}
			x += X[i];
			y += Y[i];
		}
		x -= X[idx];
		y -= Y[idx];
		printf("%.3lf %.3lf\n", x, y);
	}
    return 0;
}
