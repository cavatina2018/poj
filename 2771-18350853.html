#include <string>
#include <cstring>
using namespace std;
#define MAX_N	500
#define MAX_M	500


int N;
int heights[MAX_N];
string sex[MAX_N];
string music[MAX_N];
string sport[MAX_N];
int n, m;
int boys[MAX_N];
int girls[MAX_N];
int relations[MAX_N + MAX_M][MAX_N + MAX_M];
int relcount[MAX_N + MAX_M];
int type[MAX_N + MAX_M];
int match[MAX_N + MAX_M];
int path[MAX_N + MAX_M];
int stack[MAX_N + MAX_M];


bool findpath(int& end)
{
	int stacklen = 0;
	for (int i = 0; i < n + m; ++i)
	{
		if (type[i] == 0 && match[i] < 0)
		{
			stack[stacklen++] = i;
		}
	}

	memset(path, 0xff, sizeof(path));

	while (stacklen > 0)
	{
		int start = stack[--stacklen];

		for (int i = 0; i < relcount[start]; ++i)
		{
			end = relations[start][i];
			if (end == match[start]) continue;
			if (match[end] < 0)
			{
				path[end] = start;
				return true;
			}
			if (path[end] < 0)
			{
				path[end] = start;
				path[match[end]] = end;
				stack[stacklen++] = match[end];
			}
		}
	}

	return false;
}


int maxmatch()
{
	int res = 0, end = 0;
	if (n == 0 || m == 0)
	{
		return res;
	}

	memset(match, 0xff, sizeof(match));

	while (findpath(end))
	{
		while (true)
		{
			int prev = path[end];
			match[prev] = end;
			match[end] = prev;

			if (path[prev] < 0) break;
			end = path[prev];
		}
		++res;
	}

	return res;
}


int calc()
{
	n = m = 0;
	for (int i = 0; i < N; ++i) {
		if (sex[i] == "M") {
			boys[n++] = i;
		}
		else {
			girls[m++] = i;
		}
	}

	memset(relcount, 0, sizeof(relcount));

	for (int i = 0; i < n; ++i) {
		type[i] = 0;

		for (int j = 0; j < m; ++j)
		{
			int	fid = j + n;
			type[fid] = 1;

			int b = boys[i], g = girls[j];
			if (abs(heights[b] - heights[g]) <= 40 &&
				music[b] == music[g] &&
				sport[b] != sport[g]) {
				relations[i][relcount[i]++] = fid;
				relations[fid][relcount[fid]++] = i;
			}
		}
	}

	return N - maxmatch();
}

	int t; scanf_s("%d", &t);

			cin >> heights[j];
			cin >> sex[j];
			cin >> music[j];
			cin >> sport[j];
		}
		cout << calc() << endl;
	}
    return 0;
}
