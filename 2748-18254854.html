#include <cstring>
using namespace std;
#define MAX_N	200000
#define M		100000


int n;
int A[MAX_N];
int B[MAX_N];


void init()
{
	int a = 1, b = 1;
	for (n = 0; ; ++n) {
		A[n] = a, B[n] = b;
		int c = (a + b) % M;
		a = b, b = c;

		if (a == 1 && b == 1) {
			++n;
			break;
		}
	}
}


int calc(int x)
{
	if (x == 1) return 1;

	long long idx = (long long)2 * (x - 2) % n;
	return (A[idx] + B[idx]) % M;
}


int main()
{
	init();
	
	int t; scanf_s("%d", &t);
	for (int i = 0; i < t; ++i) {
		int x; scanf_s("%d", &x);
		printf("%d\n", calc(x));
	}
    return 0;
}
