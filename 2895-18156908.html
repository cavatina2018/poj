#include <cstring>


int keys[256];
int times[256];
int p, w;
char buf[2000];


void init()
{
	keys[' '] = 1;
	keys['A'] = keys['B'] = keys['C'] = 2;
	keys['D'] = keys['E'] = keys['F'] = 3;
	keys['G'] = keys['H'] = keys['I'] = 4;
	keys['J'] = keys['K'] = keys['L'] = 5;
	keys['M'] = keys['N'] = keys['O'] = 6;
	keys['P'] = keys['Q'] = keys['R'] = keys['S'] = 7;
	keys['T'] = keys['U'] = keys['V'] = 8;
	keys['W'] = keys['X'] = keys['Y'] = keys['Z'] = 9;
	
	times[' '] = times['A'] = times['D'] = times['G'] = times['J'] = times['M'] = times['P'] = times['T'] = times['W'] = 1;
	times['B'] = times['E'] = times['H'] = times['K'] = times['N'] = times['Q'] = times['U'] = times['X'] = 2;
	times['C'] = times['F'] = times['I'] = times['L'] = times['O'] = times['R'] = times['V'] = times['Y'] = 3;
	times['S'] = times['Z'] = 4;
}


int calc()
{
	int res = 0;
	for (int i = 0, prevkey = 0; buf[i]; ++i) {
		res += times[buf[i]] * p;
		if (keys[buf[i]] == prevkey && prevkey != 1) {
			res += w;
		}
		else {
			prevkey = keys[buf[i]];
		}
	}
	return res;
}


int main()
{
	init();

	int t; scanf_s("%d", &t);
	for (int i = 0; i < t; ++i) {
		scanf_s("%d %d", &p, &w);
		gets_s(buf, 2000);
		gets_s(buf, 2000);

		printf("%d\n", calc());
	}
    return 0;
}
