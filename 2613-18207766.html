#include <cstring>
#include <algorithm>
using namespace std;
#define MAX_N	10000


int p, q, r, s;
int arr1[MAX_N];
int arr2[MAX_N];


long double calc()
{
	if (q > p / 2) q = p - q;
	if (s > r / 2) s = r - s;

	int count = 0;
	for (int i = q, j = p; i >= 1; --i, --j) {
		arr1[count] = j;
		arr2[count] = i;
		++count;
	}
	for (int i = s, j = r; i >= 1; --i, --j) {
		arr2[count] = j;
		arr1[count] = i;
		++count;
	}

	sort(arr1, arr1 + q + s);
	sort(arr2, arr2 + q + s);

	long double res = 1.0;
	for (int i = 0; i < q + s; ++i) {
		res *= (long double)arr1[i] / arr2[i];
	}
	return res;
}


int main()
{
	while (scanf_s("%d %d %d %d", &p, &q, &r, &s) >= 4) {
		printf("%.5Lf\n", calc());
	}
    return 0;
}
