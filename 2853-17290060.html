int calc(int n)
{
	int count = 0;
	for (int i = 2; i * i <= 2 * n; ++i) {
		if (2 * n % i) continue;
		int m = 2 * n / i;
		if ((m - i + 1) % 2) continue;
		int s = (m - i + 1) / 2;
		if (s > 0) ++count;
	}
	return count;
}


int main()
{
	int t; scanf_s("%d", &t);

		printf("%d %d\n", id, calc(n));
