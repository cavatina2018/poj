#include <iostream>
#include <string>
using namespace std;


char s1[1200], s2[1200];


string calc()
{
	string res;

	int len1 = strlen(s1);
	int len2 = strlen(s2);
	int counts1[26], counts2[26];
	memset(counts1, 0, sizeof(counts1));
	memset(counts2, 0, sizeof(counts2));

	for (int i = 0; i < len1; ++i)
	{
		counts1[s1[i] - 'a']++;
	}
	for (int i = 0; i < len2; ++i)
	{
		counts2[s2[i] - 'a']++;
	}

	for (int i = 0; i < 26; ++i)
	{
		int m = counts1[i] < counts2[i] ? counts1[i] : counts2[i];
		if (m > 0)
		{
			char ch = (char)('a' + i);
			for (int j = 0; j < m; ++j)
			{
				res += ch;
			}
		}
	}
	return res;
}


int main()
{
	while (true)
	{
		if (!gets_s(s1, 1200) || !gets_s(s2, 1200))
		{
			break;
		}

		cout << calc() << endl;
	}
    return 0;
}
