#include <stdlib.h>
#include <math.h>


int calc(int n, int k)
{
	long double root = pow((long double)n, (long double)1.0 / k);
	int roots[2] = { (int)floor(root), (int)ceil(root) };

	int diff = INT_MAX, res = 0;
	for (int i = 0; i < 2; ++i) {
		int prod = 1;
		for (int j = 0; j < k; ++j) {
			prod *= roots[i];
		}
		int d = n - prod;
		if (d < 0) d = -d;
		if (d < diff) {
			diff = d;
			res = roots[i];
		}
	}

	return res;
}


int main()
{
	while (true) {
		int n, k; scanf_s("%d %d", &n, &k);
		if (!n && !k) break;
		printf("%d\n", calc(n, k));
	}
    return 0;
}
