#include <algorithm>
using namespace std;
#define MAX_N	250000

int n;
unsigned nums[MAX_N];


int main()
{
	scanf_s("%d", &n);

	for (int i = 0; i < n; ++i)
	{
		scanf_s("%u", &nums[i]);
	}
	sort(nums, nums + n);

	double res = 0;
	if (n % 2)
	{
		res = nums[n / 2];
	}
	else
	{
		res = nums[n / 2] + nums[n / 2 - 1];
		res /= 2;
	}

	printf("%.1f\n", res);
    return 0;
}
