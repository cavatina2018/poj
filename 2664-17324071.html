#include <algorithm>
using namespace std;
#define MAX_K	100
#define MAX_M	100
#define MAX_C	100


int k, m;
int course[MAX_K];
int minimal[MAX_M];
int total[MAX_M];
int category[MAX_M][MAX_C];


bool calc()
{
	sort(course, course + k);

	for (int i = 0; i < m; ++i) {
		int count = 0;
		for (int j = 0; j < total[i]; ++j) {
			int index = lower_bound(course, course + k, category[i][j]) - course;
			if (index >= k || course[index] != category[i][j]) {
				continue;
			}
			if (++count >= minimal[i]) {
				break;
			}
		}
		if (count < minimal[i]) {
			return false;
		}
	}
	return true;
}


int main()
{
	while (scanf_s("%d %d", &k, &m) > 0) {
		if (!k) break;

		for (int i = 0; i < k; ++i) {
			scanf_s("%d", &course[i]);
		}
		for (int i = 0; i < m; ++i) {
			scanf_s("%d %d", &total[i], &minimal[i]);
			for (int j = 0; j < total[i]; ++j) {
				scanf_s("%d", &category[i][j]);
			}
		}
		printf("%s\n", (calc() ? "yes" : "no"));
	}
    return 0;
}
