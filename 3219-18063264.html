#include <cstring>


long long factor2(int n)
{
	long long res = 0;
	while (n > 0) {
		n /= 2;
		res += n;
	}
	return res;
}


int calc(int n, int k)
{
	if (k == 0 || k == n) {
		return 1;
	}

	long long ret = factor2(n);
	long long ret1 = factor2(n - k);
	long long ret2 = factor2(k);

	return ret > ret1 + ret2 ? 0 : 1;
}


int main()
{
	while (true) {
		int n, k;
		if (scanf_s("%d %d", &n, &k) < 2) {
			break;
		}
		printf("%d\n", calc(n, k));
	}
    return 0;
}
