#define MAX_N	1000000


int n;
int A[MAX_N], B[MAX_N];
char results[MAX_N + 1];


int main()
{
	scanf_s("%d", &n);

	for (int i = 0; i < n; ++i)
	{
		scanf_s("%d %d", &A[i], &B[i]);
	}

	int carry = 0;

	for (int i = n - 1; i >= 0; --i)
	{
		int sum = A[i] + B[i] + carry;
		results[i] = (char)('0' + sum % 10);
		carry = sum >= 10 ? 1 : 0;
	}

	printf("%s\n", results);
    return 0;
}
