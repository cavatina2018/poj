#define SCANF	scanf
#define MAX_N	300


long long w[MAX_N + 1];


long long calc(int n)
{
	if (n == 1)
	{
		return 3;
	}

	if (w[n])
	{
		return w[n];
	}

	w[n] = n * (n + 1) * (n + 2) / 2 + calc(n - 1);
	return w[n];
}


int main()
{
	int t;
	SCANF("%d", &t);

	for (int i = 1; i <= t; ++i)
	{
		int n;
		SCANF("%d", &n);

		printf("%d %d %lld\n", i, n, calc(n));
	}
    return 0;
}
