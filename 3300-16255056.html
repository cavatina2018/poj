#include <algorithm>
using namespace std;


#define SCANF	scanf
#define MAX_F	100
#define MAX_R	100


int front[MAX_F];
int rear[MAX_R];
int f, r;
double result[MAX_F * MAX_R];


int main()
{
	while (true)
	{
		SCANF("%d", &f);
		if (!f)
		{
			break;
		}
		SCANF("%d", &r);

		for (int i = 0; i < f; ++i)
		{
			SCANF("%d", &front[i]);
		}
		for (int i = 0; i < r; ++i)
		{
			SCANF("%d", &rear[i]);
		}

		for (int i = 0; i < f; ++i)
		{
			for (int j = 0; j < r; ++j)
			{
				result[i * r + j] = (double)rear[j] / front[i];
			}
		}

		sort(result, result + r * f);

		double spread = 0;
		for (int i = 1; i < r * f; ++i)
		{
			if (result[i] / result[i - 1] > spread)
			{
				spread = result[i] / result[i - 1];
			}
		}

		printf("%.2f\n", spread);
	}
    return 0;
}
