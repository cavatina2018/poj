int grid[9][9];


bool check(int r, int c, int digit)
{
	for (int i = 0; i < 9; ++i) {
		if (grid[r][i] == digit) {
			return false;
		}
	}
	for (int i = 0; i < 9; ++i) {
		if (grid[i][c] == digit) {
			return false;
		}
	}

	int x = r / 3, y = c / 3;
	for (int i = 0; i < 3; ++i) {
		for (int j = 0; j < 3; ++j) {
			if (grid[3 * x + i][3 * y + j] == digit) {
				return false;
			}
		}
	}
	return true;
}


bool calc(int i, int j)
{
	if (j >= 9) {
		return calc(i + 1, 0);
	}
	if (i >= 9) {
		return true;
	}
	if (grid[i][j] != 0) {
		return calc(i, j + 1);
	}

	for (int k = 1; k < 10; ++k) {
		if (!check(i, j, k)) continue;
		grid[i][j] = k;
		if (calc(i, j + 1)) return true;
		grid[i][j] = 0;
	}

	return false;
}


void printres()
{
	for (int i = 0; i < 9; ++i) {
		for (int j = 0; j < 9; ++j) {
			printf("%d", grid[i][j]);
		}
		printf("\n");
	}
}

	int t; scanf_s("%d", &t);
