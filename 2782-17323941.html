#include <algorithm>
using namespace std;
#define MAX_N	100000


int n, l;
int len[MAX_N];

int main()
{
	scanf_s("%d %d", &n, &l);
	for (int i = 0; i < n; ++i) {
		scanf_s("%d", &len[i]);
	}
	sort(len, len + n);

	int index = upper_bound(len, len + n, l - len[0]) - len;
	int count = 0;

	for (int start = 0, end = index - 1; start < end; ++start, --end) {
		while (start < end && len[start] + len[end] > l) {
			--end;
		}
		if (start < end) ++count;
	}

	printf("%d\n", n - count);
    return 0;
}
