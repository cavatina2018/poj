#include <iostream>
using namespace std;


int calc(int x, int y)
{
	if (x == y) return 0;
	int n = (int)sqrt((double)y - x);
	int left = y - x - n * n;
	int res = 2 * n - 1;
	if (left % n)
	{
		res += left / n + 1;
	}
	else
	{
		res += left / n;
	}
	return res;
}


int main()
{
	int t; cin >> t;
	for (int i = 0; i < t; ++i)
	{
		int x, y; cin >> x >> y;
		cout << calc(x, y) << endl;
	}
    return 0;
}
