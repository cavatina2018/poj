#include <map>
using namespace std;

#define MAX_N	14
map<pair<int, int>, int> counts[MAX_N + 1];


void init()
{
	counts[0][make_pair(0, 0)] = 1;

	for (int i = 1; i <= MAX_N; ++i)
	{
		for (map<pair<int, int>, int>::iterator it = counts[i - 1].begin(); it != counts[i - 1].end(); ++it)
		{
			int x = it->first.first, y = it->first.second;
			counts[i][make_pair(x, y + 2)] += it->second;
			counts[i][make_pair(x, y - 2)] += it->second;
			counts[i][make_pair(x - 1, y - 1)] += it->second;
			counts[i][make_pair(x - 1, y + 1)] += it->second;
			counts[i][make_pair(x + 1, y - 1)] += it->second;
			counts[i][make_pair(x + 1, y + 1)] += it->second;
		}
	}
}


int main()
{
	init();

	int t;
	cin >> t;

	for (int i = 0; i < t; ++i)
	{
		int n;
		cin >> n;
		cout << counts[n][make_pair(0, 0)] << endl;
	}
    return 0;
}
