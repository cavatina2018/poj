#include <queue>
using namespace std;


int l, m;
queue<int> leftq, rightq;


int calc()
{
	int pos = 0, res = 0;

	while (!leftq.empty() || !rightq.empty()) {
		if (pos == 0 && !leftq.empty()) {
			int ferry = 0;
			while (!leftq.empty()) {
				if (ferry + leftq.front() > l) {
					break;
				}
				ferry += leftq.front();
				leftq.pop();
			}
		}
		else if (pos == 1 && !rightq.empty()) {
			int ferry = 0;
			while (!rightq.empty()) {
				if (ferry + rightq.front() > l) {
					break;
				}
				ferry += rightq.front();
				rightq.pop();
			}
		}
		++res;
		pos = (pos + 1) % 2;
	}

	return res;
}


int main()
{
	char buffer[10];

	int t; scanf_s("%d", &t);
	for (int i = 0; i < t; ++i) {
		scanf_s("%d %d", &l, &m);
		l *= 100;

		for (int j = 0; j < m; ++j) {
			int car; scanf_s("%d %s", &car, buffer, 10);
			if (!strcmp(buffer, "left")) {
				leftq.push(car);
			}
			else {
				rightq.push(car);
			}
		}
		printf("%d\n", calc());
	}
    return 0;
}
