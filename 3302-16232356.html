bool subsequence(string& a, string& b)
{
	int i = 0, j = 0;
	bool flag = true;

	for (; i < b.length(); ++i)
	{
		while (j < a.length() && a[j] != b[i])
		{
			++j;
		}
		if (j >= a.length())
		{
			flag = false;
			break;
		}
		++j;
	}

	if (flag)
	{
		return true;
	}

	i = b.length() - 1, j = 0;
	for (; i >= 0; --i)
	{
		while (j < a.length() && a[j] != b[i])
		{
			++j;
		}
		if (j >= a.length())
		{
			return false;
		}
		++j;
	}

	return true;
}
