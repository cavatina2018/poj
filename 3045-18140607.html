#include <cstring>
#include <algorithm>
using namespace std;
#define MAX_N	60000


int n;
long long W[MAX_N];
long long S[MAX_N];
int ranks[MAX_N];


bool Less(int a, int b)
{
	return W[a] + S[a] < W[b] + S[b];
}


long long calc()
{
	sort(ranks, ranks + n, Less);

	long long res = LLONG_MIN;
	for (long long i = 0, sum = 0; i < n; ++i) {
		int idx = ranks[i];
		long long ret = sum - S[idx];
		if (ret > res) res = ret;
		sum += W[idx];
	}

	return res;
}


int main()
{
	scanf_s("%d", &n);
	for (int i = 0; i < n; ++i) {
		scanf_s("%lld %lld", &W[i], &S[i]);
		ranks[i] = i;
	}
	printf("%lld\n", calc());
    return 0;
}
