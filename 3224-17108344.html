#define MAX_N	100


int lostcount[MAX_N];

int main()
{
	int n; scanf_s("%d", &n);

	int best = 0, index = 0;
	for (int i = 0; i < n; ++i) {
		int count = 0;
		bool flag = true;

		for (int j = 0; j < n; ++j) {
			int a; scanf_s("%d", &a);
			if (a == 3) {
				count++;
				lostcount[j]++;
			}
			if (lostcount[j] < n - 1 - best) {
				flag = false;
			}
		}
		if (flag) {
			printf("%d\n", index + 1);
			return 0;
		}
		if (count > best) {
			best = count;
			index = i;
		}
	}
	printf("%d\n", index + 1);
    return 0;
}
