#include <cstring>
#include <algorithm>
using namespace std;
#define MAX_N	500000


int n;
char str[MAX_N];
int counts[MAX_N];
int rescount;
int results[MAX_N];


void calc()
{
	n = strlen(str);
	counts[1] = 0;
	rescount = 0;

	for (int i = 1, curr = 0; i < n; ++i) {
		while (str[curr] != str[i]) {
			if (curr == 0) break;
			curr = counts[curr];
		}
		if (str[curr] != str[i]) {
			counts[i + 1] = curr;
		}
		else {
			counts[i + 1] = ++curr;
		}
	}

	for (int i = counts[n]; i > 0; i = counts[i]) {
		results[rescount++] = i;
	}
	results[rescount++] = n;

	sort(results, results + rescount);

	for (int i = 0; i < rescount; ++i) {
		printf("%d ", results[i]);
	}
	printf("\n");
}


int main()
{
	while (scanf_s("%s", str, MAX_N) > 0) {
		calc();
	}
    return 0;
