#include <algorithm>
using namespace std;
#define MAX_N	100000


int n;
long long T[MAX_N];
long long D[MAX_N];
int ranking[MAX_N];


bool Less(int a, int b)
{
	return D[a] * T[b] > D[b] * T[a];
}


long long calc()
{
	sort(ranking, ranking + n, Less);

	long long res = 0;
	for (long long i = 0, start = 0; i < n; ++i) {
		res += start * D[ranking[i]];
		start += T[ranking[i]] * 2;
	}

	return res;
}


int main()
{
	scanf_s("%d", &n);
	for (int i = 0; i < n; ++i) {
		scanf_s("%lld %lld", &T[i], &D[i]);
		ranking[i] = i;
	}
	printf("%lld\n", calc());
    return 0;
}
