#include <cstring>
#define MAX_N	10000


int n;
int nums[MAX_N];


int calc()
{
	int sum = 0;
	for (int i = 0; i < n; ++i) {
		sum ^= nums[i];
	}

	if (sum == 0) return 0;

	int k = 0;
	for (int i = 0; i < 32; ++i) {
		if (sum & (1 << i)) {
			k = i;
		}
	}

	int res = 0;
	for (int i = 0; i < n; ++i) {
		if (nums[i] & (1 << k)) {
			++res;
		}
	}

	return res;
}


int main()
{
	while (true) {
		scanf_s("%d", &n);
		if (!n) break;

		for (int i = 0; i < n; ++i) {
			scanf_s("%d", &nums[i]);
		}

		printf("%d\n", calc());
	}
    return 0;
}
