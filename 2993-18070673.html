void add(char chess, int col, int row, bool white)
{
	if (!white) {
		chess += 'a' - 'A';
	}
	row = 7 - row;

	board[2 * row + 1][4 * col + 2] = chess;
}


void process(char* buf, bool white)
{
	for (int i = 0; buf[i]; ) {
		if (buf[i] == ',') {
			i++;
		}
		else if (buf[i] >= 'A' && buf[i] <= 'Z') {
			add(buf[i], buf[i + 1] - 'a', buf[i + 2] - '1', white);
			i += 3;
		}
		else {
			add('P', buf[i] - 'a', buf[i + 1] - '1', white);
			i += 2;
		}
	}
}

	process(buf2, false);
