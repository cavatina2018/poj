#include <map>
using namespace std;


int main()
{
	map<int, int> factors;

	for (int i = 0; i < 10; ++i)
	{
		int n; cin >> n;

		for (int j = 2; j * j <= n; )
		{
			while (n % j == 0)
			{
				factors[j]++;
				n /= j;
			}
			++j;
		}
		if (n > 1) factors[n]++;
	}

	int res = 1;
	for (map<int, int>::iterator it = factors.begin(); it != factors.end(); ++it)
	{
		res *= it->second + 1;
		res %= 10;
	}

	cout << res << endl;
    return 0;
}
