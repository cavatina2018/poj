#define SCANF	scanf
#define MAX_K	10

int rooms[MAX_K];


int main()
{
	int n, m, k;
	SCANF("%d %d %d", &n, &m, &k);
	for (int i = 0; i < k; ++i)
	{
		SCANF("%d", &rooms[i]);
	}

	int count = 0;

	for (int i = 0; i < k; ++i)
	{
		int c;
		SCANF("%d", &c);
		if (c > rooms[i])
		{
			count += c - rooms[i];
		}
	}
	printf("%d", count);
    return 0;
}
