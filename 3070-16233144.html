#include <vector>
#include <set>
using namespace std;

#define SCANF	scanf
set<pair<int, int> > s;
vector<int> v;


void init()
{
	int prev1 = 0, prev2 = 1;
	while(true)
	{
		if (s.find(make_pair(prev1, prev2)) != s.end())
		{
			break;
		}
		int curr = (prev1 + prev2) % 10000;
		v.push_back(curr);
		s.insert(make_pair(prev1, prev2));
		prev1 = prev2, prev2 = curr;
	}
}


int calc(long long n)
{
	if (n == 0) return 0;
	if (n == 1) return 1;
	return v[(n - 2) % v.size()];
}


int main()
{
	init();
	while (true)
	{
		long long n;
		SCANF("%lld", &n);
		if (n == -1)
		{
			break;
		}
		printf("%d\n", calc(n));
	}
    return 0;
}
