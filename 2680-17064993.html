import java.util.*;
import java.io.*;
import java.lang.*;


public class Main {
	
	public static final int MAX_N = 1000;
	public static boolean[] states = new boolean[MAX_N + 1];
	public static BigInteger[] f = new BigInteger[MAX_N + 1];
	public static BigInteger[] g = new BigInteger[MAX_N + 1];
	
	
	static void calc(int n)
	{
		if (states[n]) return;

		calc(n - 1);
		g[n] = f[n - 1].add(g[n - 1]);
		f[n] = f[n - 1].add(g[n - 1]);
		if (n % 2 == 0) f[n] = f[n].add(BigInteger.ONE);

		states[n] = true;
	}
	
	
	public static void main(String[] args) {
        Scanner in = new Scanner(System.in);
		
		states[0] = true;
		f[0] = BigInteger.ZERO;
		g[0] = BigInteger.ZERO;

		while(in.hasNext()) {
			int n = in.nextInt();
			calc(n);
			System.out.println(f[n]);
		}
	}
};
