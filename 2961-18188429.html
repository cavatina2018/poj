long long n, x, y, w, h;


int calc(long long level, long long r, long long c)
{
	if (level <= 1) return 1;
	level /= 2;
	if (r >= level && c >= level) {
		return -calc(level, r % level, c % level);
	}
	return calc(level, r % level, c % level);
}


void calc()
{
	for (int i = 0; i < h; ++i) {
		for (int j = 0; j < w; ++j) {
			printf("%d ", calc(n, y + i, x + j));
		}
		printf("\n");
	}
	printf("\n");
}
