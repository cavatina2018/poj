#include <math.h>
#define SCANF	scanf
#define MAX_PRIME	200000


bool prime[MAX_PRIME + 1];


void init()
{
	prime[1] = false;
	prime[2] = true;

	for (int i = 3; i <= MAX_PRIME; ++i)
	{
		bool flag = true;
		for (int j = 2; j <= sqrt((double)i); ++j)
		{
			if (i % j == 0)
			{
				flag = false;
				break;
			}
		}
		if (flag)
		{
			prime[i] = true;
		}
	}
}


int calc(int n)
{
	int res = 0;

	for (int i = 1; i <= n / 2; ++i)
	{
		if (prime[i] && prime[n - i])
		{
			res++;
		}
	}

	return res;
}

int main()
{
	init();
	while (true)
	{
		int n; SCANF("%d", &n);
		if (!n) break;
		printf("%d\n", calc(n));
	}
    return 0;
}
