#include <string>
#include <set>
using namespace std;


bool surprising(string& s)
{
	if (s.length() <= 2)
	{
		return true;
	}
	for (int i = 0; i <= s.length() - 2; ++i)
	{
		set<int> cache;

		for (size_t j = 0; j <= s.length() - 2 - i; ++j)
		{
			int a = ((int)s[j]) << 8 | s[j + 1 + i];
			if (cache.find(a) != cache.end())
			{
				return false;
			}
			cache.insert(a);
		}
	}
	return true;
}

int main()
{
	while (true)
	{
		string s;
		cin >> s;
		if (s == "*")
		{
			break;
		}

		if (surprising(s))
		{
			cout << s << " is surprising." << endl;
		}
		else
		{
			cout << s << " is NOT surprising." << endl;
		}
	}
    return 0;
}
