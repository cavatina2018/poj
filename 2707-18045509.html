#include <cstring>


int a, b, c, d;


int calc()
{
	if (a <= c && b <= d || a <= d && b <= c) {
		return 100;
	}

	double r1, r2;
	{
		double ret1 = 100.0 * c / a;
		double ret2 = 100.0 * d / b;
		r1 = ret1 < ret2 ? ret1 : ret2;
	}
	{
		double ret1 = 100.0 * d / a;
		double ret2 = 100.0 * c / b;
		r2 = ret1 < ret2 ? ret1 : ret2;
	}

	double r = r1 > r2 ? r1 : r2;
	return (int)r;
}


int main()
{
	while (true) {
		scanf_s("%d %d %d %d", &a, &b, &c, &d);
		if (!a && !b && !c && !d) break;
		printf("%d%%\n", calc());
	}
    return 0;
}
