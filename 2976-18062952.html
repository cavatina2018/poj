#include <algorithm>
using namespace std;
#define MAX_N	2000


int n, k;
int A[MAX_N];
int B[MAX_N];
double x;
int ranks[MAX_N];


bool Less(int a, int b)
{
	return A[a] - x * B[a] > A[b] - x * B[b];
}


bool test()
{
	sort(ranks, ranks + n, Less);

	double sum = 0.0;
	for (int i = 0; i < n - k; ++i) {
		int idx = ranks[i];
		sum += A[idx] - x * B[idx];
	}

	return sum >= 0.0;
}


int calc()
{
	for (int i = 0; i < n; ++i) {
		ranks[i] = i;
	}

	double a = 0.0, b = 1.0, res = a;
	while (true) {
		if (b - a < 0.00001) {
			break;
		}
		x = (a + b) / 2.0;
		if (test()) {
			if (x > res) res = x;
			a = x;
		}
		else {
			b = x;
		}
	}

	return (int)(res * 100.0 + 0.5);
}


int main()
{
	while (true) {
		scanf_s("%d %d", &n, &k);
		if (!n && !k) break;

		for (int i = 0; i < n; ++i) {
			scanf_s("%d", &A[i]);
		}
		for (int i = 0; i < n; ++i) {
			scanf_s("%d", &B[i]);
		}
		
		printf("%d\n", calc());
	}
    return 0;
}
