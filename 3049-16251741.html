#include <iostream>
#include <set>
#include <string>
#include <algorithm>
using namespace std;


int c, l;
int alphabets[26];
set<string> results;
char res[16];


void generate(int index, int len, int maxlen, int vowels, int consonants)
{
	if (maxlen - len > c - index)
	{
		return;
	}

	if (maxlen <= len)
	{
		if (vowels >= 1 && consonants >= 2)
		{
			results.insert(res);
		}
		return;
	}

	generate(index + 1, len, maxlen, vowels, consonants);

	char ch = alphabets[index];
	res[len] = ch;
	if (ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u')
	{
		generate(index + 1, len + 1, maxlen, vowels + 1, consonants);
	}
	else
	{
		generate(index + 1, len + 1, maxlen, vowels, consonants + 1);
	}
	res[len] = 0;
}


int main()
{
	cin >> l >> c;
	for (int i = 0; i < c; ++i)
	{
		string line; cin >> line;
		alphabets[i] = line[0];
	}
	sort(alphabets, alphabets + c);

	generate(0, 0, l, 0, 0);

	for (set<string>::iterator it = results.begin(); it != results.end(); ++it)
	{
		printf("%s\n", it->c_str());
	}
    return 0;
}
