#include <string>
using namespace std;
#define MAX_N	20


string src[MAX_N + 1];
bool srcstate[MAX_N + 1];
string dst[MAX_N + 1];
bool dststate[MAX_N + 1];


int main()
{
	char line[10];

	int t; scanf_s("%d", &t);
	for (int i = 0; i < t; ++i) {
		memset(srcstate, 0, sizeof(srcstate));
		memset(dststate, 0, sizeof(dststate));

		int n, m; scanf_s("%d %d", &n, &m);
		for (int j = 1; j <= n; ++j) {
			scanf_s("%s", line, 10);
			src[j] = line;
		}

		for (int j = 0; j < m; ++j) {
			int from, to; scanf_s("%d %d", &from, &to);
			dst[to] = src[from];
			dststate[to] = srcstate[from] = true;
		}

		for (int s = 1, t = 1; s <= n; ++s) {
			if (srcstate[s]) continue;
			for (; t <= n; ++t) {
				if (dststate[t]) continue;
				srcstate[s] = dststate[t] = true;
				dst[t] = src[s];
				break;
			}
		}

		for (int j = 1; j <= n; ++j) {
			printf("%s ", dst[j].c_str());
		}
		printf("\n");
	}
    return 0;
}
