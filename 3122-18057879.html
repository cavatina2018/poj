#include <float.h>
#include <cstring>
#define MAX_N	10000
#define PI		3.1415926535898
#define S		1000000


int n, f;
long long cakes[MAX_N];
long long m;


bool test(long long x)
{
	long long count = 0;
	for (int i = 0; i < n && count < f; ++i) {
		count += cakes[i] / x;
	}
	return count >= f;
}


long long calc()
{
	long long a = 1, b = m, res = 0;
	while (a <= b) {
		long long c = (a + b) / 2;
		if (test(c)) {
			if (c > res) res = c;
			a = c + 1;
		}
		else {
			b = c - 1;
		}
	}
	return res;
}


int main()
{
	int t; scanf_s("%d", &t);
	for (int i = 0; i < t; ++i) {
		scanf_s("%d %d", &n, &f);
		while (n < 0 || n > MAX_N) printf("...");
		++f;
		m = 0;

		for (int j = 0; j < n; ++j) {
			int r; scanf_s("%d", &r);
			cakes[j] = (long long)(PI * r * r * S);
			if (cakes[j] > m) {
				m = cakes[j];
			}
		}

		printf("%.4lf\n", (double)calc() / S);
	}
    return 0;
}
