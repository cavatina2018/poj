using namespace std;


#define MAX_INT		0x7fffffff
#define MAX_N	100000
int N, S;
int Numbers[MAX_N];


int calc()
{
	int sum = 0, begin = 0, end = 0, res = MAX_INT;

	while(end < N || sum >= S)
	{
		if (sum < S)
		{
			sum += Numbers[end++];
			if (sum >= S)
			{
				int count = end - begin;
				if (count == 1)
				{
					return 1;
				}
				if (count < res)
				{
					res = count;
				}
			}
		}
		else
		{
			sum -= Numbers[begin++];
			if (sum >= S)
			{
				int count = end - begin;
				if (count == 1)
				{
					return 1;
				}
				if (count < res)
				{
					res = count;
				}
			}
		}
	}

	if (res == MAX_INT)
	{
		res = 0;
	}
	return res;
}


int main()
{
	int t; cin >> t;
	for (int i = 0; i < t; ++i)
	{
		cin >> N >> S;
		for (int j = 0; j < N; ++j)
		{
			cin >> Numbers[j];
		}
		cout << calc() << endl;
	}
    return 0;
}
