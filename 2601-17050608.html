#include <iostream>
using namespace std;
#define MAX_N	3000


int n;
double cost[MAX_N + 1];


int main()
{
	cin >> n;

	double s1, s2; cin >> s1 >> s2;

	for (int i = 1; i <= n; ++i)
	{
		cin >> cost[i];
	}

	double a = 1.0, b = 0.0, c = 0.0, d = s1;

	for (int i = 2; i <= n + 1; ++i)
	{
		double e = a * 2 - c;
		double f = 2 * (b + cost[i - 1]) - d;
		c = a, d = b, a = e, b = f;
	}

	double res = (s2 - b) / a;
	printf("%.2f\n", res);
    return 0;
}
