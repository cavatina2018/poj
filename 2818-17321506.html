#include <stdlib.h>
#define min(a, b)	((a) < (b) ? (a) : (b))


int main()
{
	while (true) {
		int q, d, n, p, c;
		scanf_s("%d %d %d %d %d", &q, &d, &n, &p, &c);
		if (!q && !d && !n && !p && !c) break;

		int result = INT_MAX;
		int savecounts[4];

		for (int i = 0; i <= q; ++i) {
			if (i * 25 > c) break;

			int res = i;
			int counts[4] = { i };

			int left = c - i * 25;
			if(left > 0)
			{
				counts[1] = min(left / 10, d);
				left -= counts[1] * 10;
				res += counts[1];
			}
			if(left > 0)
			{
				counts[2] = min(left / 5, n);
				left -= counts[2] * 5;
				res += counts[2];
			}
			if (left > 0)
			{
				counts[3] = min(left, p);
				left -= counts[3];
				res += counts[3];
			}

			if (left == 0) {
				if (res < result) {
					result = res;
					savecounts[0] = counts[0];
					savecounts[1] = counts[1];
					savecounts[2] = counts[2];
					savecounts[3] = counts[3];
				}
			}
		}

		if (result == INT_MAX) {
			printf("Cannot dispense the desired amount.\n");
		}
		else {
			printf("Dispense %d quarters, %d dimes, %d nickels, and %d pennies.\n",
				savecounts[0], savecounts[1], savecounts[2], savecounts[3]);
		}
	}
    return 0;
}
