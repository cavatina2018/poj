void init()
{
	for (int i = 0; i < sizeof(initdata) / sizeof(Result); ++i) {
		cache[initdata[i].digits] = initdata[i].results;

}

{
	init();

	char buffer[1024];

	int t; cin >> t;
	cin.getline(buffer, 1024);

	for (int i = 0; i < t; ++i)
	{
		cin.getline(buffer, 1024);
		stringstream ss; ss.str(buffer);
		for (n = 0; ss >> digits[n]; ++n);

		sort(digits, digits + n);

		string s;
		for (int j = 0; j < n; ++j) {

		cout << cache[s] << endl;
	}
