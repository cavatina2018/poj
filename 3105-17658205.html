long double calc(long long n)
{
	long double res = 0.0;

	for (int i = 1; ; ++i) {
		long long bit = 1LL << i;
		long long zero = n / bit * (bit / 2), one = n / bit * (bit / 2);
		long long left = n % bit;
		if (left <= bit / 2) {
			zero += left;
		}
		else {
			zero += bit / 2;
			one += left - bit / 2;
		}
		res += (long double)one * zero * bit / n / n;

		if (bit > n) break;
	}

	return (long double)res;
}
