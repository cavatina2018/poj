#include <string>
using namespace std;


long long quicksum(string& s)
{
	long long sum = 0;
	for (size_t i = 0; i < s.length(); ++i)
	{
		if (s[i] == ' ')
		{
			continue;
		}
		sum += (i + 1) * (s[i] - 'A' + 1);
	}
	return sum;
}

int main()
{
	while (true)
	{
		char line[256];
		cin.getline(line, 256);

		string s(line);
		if (s == "#")
		{
			break;
		}
		cout << quicksum(s) << endl;
	}
    return 0;
}
