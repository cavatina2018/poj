using namespace std;


void calc(int n, vector<pair<int, int> >& res)
{
	if (n == 0) {
		return;
	}

	if (n == 1) {
		res.push_back(make_pair(0, 0));
		return;
	}

	if (n % 3 == 0) {
		vector<pair<int, int> > res2;
		calc(n / 3, res2);
		for (size_t i = 0; i < res2.size(); ++i) {
			res.push_back(make_pair(res2[i].first, res2[i].second + 1));
		}
		return;
	}

	for (int i = 30; i >= 0; --i) {
		int a = 1 << i;
		if (a <= n && (n - a) % 3 == 0) {
			vector<pair<int, int> > res2;
			calc(n - a, res2);
			for (size_t i = 0; i < res2.size(); ++i) {
				res.push_back(res2[i]);
			}
			res.push_back(make_pair(i, 0));
			return;
		}
	}

	while (true) printf("...");
}


void printres(int t, vector<pair<int, int> >& res)
{
	printf("%d %lu ", t, res.size());

	for (int i = res.size() - 1; i >= 0; --i) {
		printf("[%d,%d] ", res[i].first, res[i].second);
	}

	printf("\n");
}


int main()
{
	int t; scanf_s("%d", &t);
	for (int i = 1; i <= t; ++i) {
		int n; scanf_s("%d", &n);
		vector<pair<int, int> > res;
		calc(n, res);
		printres(i, res);
	}
    return 0;
