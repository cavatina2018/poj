#include <cstdlib>
#include <cmath>
#include <algorithm>
using namespace std;
#define MAX_N	100000


int n;
long long m;
int values[MAX_N + 1];
long long sums[MAX_N + 1];
long long results[MAX_N + 1];
int indices[MAX_N + 1];
int tops[MAX_N + 1];
int indicecount;


long long calc()
{
	sums[0] = 0;
	for (int i = 1; i <= n; ++i) {
		sums[i] = sums[i - 1] + values[i];
	}

	results[0] = 0;
	indices[0] = 0;
	indicecount = 1;

	for (int i = 1; i <= n; ++i) {
		int idx = indices[indicecount - 1];
		long long res = results[idx] + values[i];
		for (int j = indicecount - 2; j >= 0; --j) {
			idx = indices[j];
			if (sums[i] - sums[idx] > m) break;
			tops[j] = max(tops[j], values[i]);
			long long ret = results[idx] + tops[j];
			if (ret < res) res = ret;
		}
		results[i] = res;
		if (res != results[indices[indicecount - 1]]) {
			tops[indicecount - 1] = values[i];
			indices[indicecount++] = i;
		}
		else {
			indices[indicecount - 1] = i;
		}
	}

	return results[n];
}


int main()
{
	scanf_s("%d %lld", &n, &m);
	for (int i = 1; i <= n; ++i) {
		scanf_s("%d", &values[i]);
		if (values[i] > m) {
			printf("-1\n");
			return 0;
		}
	}

	printf("%lld\n", calc());
	return 0;
}
