#include <cstring>
#include <algorithm>
using namespace std;
#define MAX_N	100000


int n;
int id[MAX_N];
int len[MAX_N];
double freq[MAX_N];
int ranks[MAX_N];
int idx;


bool Less(int a, int b)
{
	return freq[a] * len[b] > freq[b] * len[a];
}


int calc()
{
	sort(ranks, ranks + n, Less);
	return id[ranks[idx - 1]];
}


int main()
{
	while (scanf_s("%d", &n) >= 1) {
		for (int i = 0; i < n; ++i) {
			scanf_s("%d %d %lf", &id[i], &len[i], &freq[i]);
			ranks[i] = i;
		}
		scanf_s("%d", &idx);

		printf("%d\n", calc());
	}
    return 0;
}
