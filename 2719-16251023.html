#define SCANF	scanf


int calc(int n)
{
	int res = 0, weight = 1;

	while (n > 0)
	{
		int i = n % 10;
		res += (i > 4 ? i - 1 : i) * weight;
		n /= 10;
		weight *= 9;
	}
	return res;
}


int main()
{
	while (true)
	{
		int n; SCANF("%d", &n);
		if (!n) break;
		printf("%d: %d\n", n, calc(n));
	}
    return 0;
}
