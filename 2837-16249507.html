#include <stdio.h>
#define SCANF	scanf
#define MAX_N	512


int set[MAX_N * 2];
int grid[MAX_N + 1][MAX_N + 1];


void calc(int i, int j, int n, int* s)
{
	if (n == 2)
	{
		grid[i][j] = s[0];
		grid[i + 1][j + 1] = s[0];
		grid[i + 1][j] = s[1];
		grid[i][j + 1] = s[2];
	}
	else
	{
		calc(i, j, n / 2, s);
		calc(i + n / 2, j + n / 2, n / 2, s);
		calc(i, j + n / 2, n / 2, s + n);
		int temp = s[n];
		s[n] = s[n - 1];
		calc(i + n / 2, j, n / 2, s + n);
		s[n] = temp;
	}
}


void print_matrix(int n)
{
	for (int i = 1; i <= n; ++i)
	{
		for (int j = 1; j <= n; ++j)
		{
			printf("%d\t", grid[i][j]);
		}
		printf("\n");
	}
}


int main()
{
	int k; SCANF("%d", &k);
	int n = (int)pow(2.0, k);

	for (int i = 1; i <= 2 * n - 1; ++i)
	{
		set[i] = i;
	}

	int* s = &set[1];
	calc(1, 1, n, s);
	print_matrix(n);
    return 0;
}
