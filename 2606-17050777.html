#include <math.h>

using namespace std;
#define MAX_N	700


int n;
int xarr[MAX_N], yarr[MAX_N];
int cachecount;
unsigned long long cache[MAX_N];



int gcd(int a, int b)
{
	if (a < b)
	{
		if (b % a == 0)
		{
			return a;
		}
		return gcd(b - a, a);
	}
	else if (a > b)
	{
		if (a % b == 0)
		{
			return b;
		}
		return gcd(a - b, b);
	}

	return a;
}


unsigned long long make_key(int x, int y)
{
	bool s1 = false, s2 = false;
	if (x < 0)
	{
		s1 = true;
		x = -x;
	}
	if (y < 0)
	{
		s2 = true;
		y = -y;
	}

	unsigned long long res = 0;
	if (y == 0)
	{
		res = 0;
	}
	else if (x == 0)
	{
		res = (unsigned long long) - 1ll;
	}
	else
	{
		int m = gcd(x, y);
		x /= m;
		y /= m;

		res = ((unsigned long long)x << 32) | y;
		if (s1 && !s2 || !s1 && s2)
		{
			res |= 0x8000000000000000;
		}
	}

	return res;
}


int calc()
{
	if (n == 1 || n == 2)
	{
		return n;
	}

	int m = 2;

	for (int i = 1; i < n; ++i)
	{
		cachecount = 0;

		for (int j = 0; j < i; ++j)
		{
			int x = xarr[i] - xarr[j];
			int y = yarr[i] - yarr[j];
			unsigned long long key = make_key(x, y);
			cache[cachecount++] = key;
		}

		sort(cache, cache + cachecount);

		int s = 2;
		unsigned long long v = cache[0];
		for (int j = 1; j < cachecount; ++j)
		{
			if (cache[j] == v)
			{
				s++;
			}
			else
			{
				if (s > m)
				{
					m = s;
				}
				s = 2;
				v = cache[j];
			}
		}
		if (s > m)
		{
			m = s;
		}
	}

	return m;
}
