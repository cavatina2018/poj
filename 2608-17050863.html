#include <sstream>
#include <string>
using namespace std;


int mapping[256];


void init()
{
	char *alpha[] = {
		"BFPV",
		"CGJKQSXZ",
		"DT",
		"L",
		"MN",
		"R"
	};

	for (int i = 0; i < 6; ++i)
	{
		for (int j = 0; alpha[i][j]; ++j)
		{
			mapping[alpha[i][j]] = i + 1;
		}
	}
}


int main()
{
	init();

	string s;
	while (cin >> s)
	{
		stringstream ss;
		int prev = 0;

		for (int i = 0; i < s.length(); ++i)
		{
			int curr = mapping[s[i]];
			if (curr != 0 && curr != prev)
			{
				ss << curr;
			}
			prev = curr;
		}

		cout << ss.str() << endl;
	}
    return 0;
}
