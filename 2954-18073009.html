#include <math.h>


int X[3], Y[3];


double S()
{
	double a = sqrt((double)(X[0] - X[1]) * (X[0] - X[1]) + (Y[0] - Y[1]) * (Y[0] - Y[1]));
	double b = sqrt((double)(X[0] - X[2]) * (X[0] - X[2]) + (Y[0] - Y[2]) * (Y[0] - Y[2]));
	double c = sqrt((double)(X[2] - X[1]) * (X[2] - X[1]) + (Y[2] - Y[1]) * (Y[2] - Y[1]));
	double d = (a + b + c) / 2;
	double s = sqrt(d * (d - a) * (d - b) * (d - c));
	return s;
}


int gcd(int m, int n)
{
	if (m > n) {
		if (m % n == 0) return n;
		return gcd(n, m % n);
	}
	else {
		if (n % m == 0) return m;
		return gcd(m, n % m);
	}
}


int calc(int dx, int dy)
{
	if (dx == 0) return abs(dy) - 1;
	if (dy == 0) return abs(dx) - 1;
	return gcd(abs(dx), abs(dy)) - 1;
}


int calc()
{
	double edgepoints = 3;
	edgepoints += calc(X[0] - X[1], Y[0] - Y[1]);
	edgepoints += calc(X[0] - X[2], Y[0] - Y[2]);
	edgepoints += calc(X[2] - X[1], Y[2] - Y[1]);

	double s = S();

	return (int)(s + 1 - edgepoints / 2 + 0.5);
}
